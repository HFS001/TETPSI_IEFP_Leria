Algoritmo - TabuadaInício      Escreva "Digite um número:"      Leia número      Para i de 1 até 10 faça          Resultado ? número * i          Escreva número, " x ", i, " = ", Resultado      FimPara  Fim#include <stdio.h>int main() {    int numero, i;        printf("Digite um número: ");    scanf("%d", &numero);    for (i = 1; i <= 10; i++) {        printf("%d x %d = %d\n", numero, i, numero * i);    }    return 0;}
